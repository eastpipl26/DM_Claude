## 크롤링 실행 방법

API Key 확인 완료 후 아래 Python 코드를 Bash로 실행한다.

### 마크다운 크롤링 (기본)

```python
from firecrawl import FirecrawlApp

app = FirecrawlApp(api_key="<firecrawl.env에서 읽은 키>")
result = app.scrape_url("<대상 URL>", formats=["markdown"])
print(result.get("markdown", "마크다운 변환 실패"))
```

### 스크린샷 캡처

URL 캡처가 필요할 때 `formats=["screenshot"]`을 사용한다.
결과의 `screenshot` 필드에 base64 인코딩 PNG가 반환된다.
저장 경로는 호출한 에이전트가 지정하며, 기본값은 `C:\Users\eastp\.claude\assets\screenshots\`.

```python
import base64, os, time
from firecrawl import FirecrawlApp

app = FirecrawlApp(api_key="<firecrawl.env에서 읽은 키>")
result = app.scrape_url("<대상 URL>", formats=["screenshot"])

screenshot_b64 = result.get("screenshot", "")
if screenshot_b64:
    # base64 디코딩 → PNG 저장
    os.makedirs(r"C:\Users\eastp\.claude\assets\screenshots", exist_ok=True)
    filename = f"screenshot_{int(time.time())}.png"
    filepath = rf"C:\Users\eastp\.claude\assets\screenshots\{filename}"
    with open(filepath, "wb") as f:
        f.write(base64.b64decode(screenshot_b64))
    print(f"저장 완료: {filepath}")
else:
    print("스크린샷 실패")
```

### 이미지 직접 다운로드

특정 이미지 URL을 파일로 저장할 때 사용한다.
저장 경로는 호출한 에이전트가 지정하며, 기본값은 `C:\Users\eastp\.claude\assets\images\`.

```python
import urllib.request
import os
import time
from pathlib import Path

def download_image(image_url: str, save_dir: str = r"C:\Users\eastp\.claude\assets\images", filename: str = None) -> str:
    """
    이미지 URL을 직접 다운로드해 저장한다.
    filename 미지정 시 URL에서 자동 추출, 중복 방지용 타임스탬프 접두사 추가.
    반환값: 저장된 파일 경로
    """
    os.makedirs(save_dir, exist_ok=True)

    if not filename:
        url_filename = image_url.split("?")[0].rstrip("/").split("/")[-1]
        ext = Path(url_filename).suffix or ".jpg"
        filename = f"{int(time.time())}_{url_filename if url_filename else 'image' + ext}"

    save_path = os.path.join(save_dir, filename)

    req = urllib.request.Request(image_url, headers={"User-Agent": "Mozilla/5.0"})
    with urllib.request.urlopen(req, timeout=30) as resp:
        with open(save_path, "wb") as f:
            f.write(resp.read())

    return save_path

# 사용 예시
path = download_image("https://example.com/photo.jpg")
print(f"저장 완료: {path}")
```

#### 페이지 내 이미지 전체 일괄 다운로드

마크다운 크롤링 결과에서 이미지 URL을 추출해 전부 저장한다.

```python
import re

def download_images_from_markdown(markdown: str, save_dir: str = r"C:\Users\eastp\.claude\assets\images") -> list:
    """
    마크다운 본문에서 이미지 URL을 추출해 전부 다운로드한다.
    반환값: 저장된 파일 경로 목록
    """
    urls = re.findall(r'!\[.*?\]\((https?://[^\)]+)\)', markdown)
    saved = []
    for url in urls:
        try:
            path = download_image(url, save_dir)
            saved.append(path)
            print(f"✓ {path}")
        except Exception as e:
            print(f"✗ {url} — {e}")
    return saved
```
